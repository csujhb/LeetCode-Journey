## 7. C璇█浠ｇ爜瀹炵幇

```c
#include <stdlib.h>
#include <string.h>
#include <limits.h>

char* minWindow(char* s, char* t) {
    int lenS = strlen(s);
    int lenT = strlen(t);

    if (lenS == 0 || lenT == 0 || lenS < lenT) {
        char* ans = (char*)malloc(1);
        ans[0] = '\0';
        return ans;
    }

    int need[128] = {0};
    int window[128] = {0};

    int required = 0;
    for (int i = 0; i < lenT; i++) {
        if (need[(unsigned char)t[i]] == 0) {
            required++;
        }
        need[(unsigned char)t[i]]++;
    }

    int formed = 0;
    int left = 0;
    int minLen = INT_MAX;
    int start = 0;

    for (int right = 0; right < lenS; right++) {
        unsigned char c = (unsigned char)s[right];
        window[c]++;

        if (need[c] > 0 && window[c] == need[c]) {
            formed++;
        }

        while (left <= right && formed == required) {
            if (right - left + 1 < minLen) {
                minLen = right - left + 1;
                start = left;
            }

            unsigned char d = (unsigned char)s[left];
            window[d]--;

            if (need[d] > 0 && window[d] < need[d]) {
                formed--;
            }

            left++;
        }
    }

    if (minLen == INT_MAX) {
        char* ans = (char*)malloc(1);
        ans[0] = '\0';
        return ans;
    }

    char* ans = (char*)malloc(minLen + 1);
    strncpy(ans, s + start, minLen);
    ans[minLen] = '\0';
    return ans;
}
```
